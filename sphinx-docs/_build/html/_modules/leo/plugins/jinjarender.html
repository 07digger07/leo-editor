[docs]def jinja_act_on_node(c,p, event):
    h = p.h
    
    #print "try act"
    if not h.startswith('@jinja '):
        raise leoPlugins.TryNext
    
    #print "act"
    tail = h[7:].strip()
    pth = c.getNodePath(p)
    fullpath = g.os_path_finalize_join(pth, tail)
    g.es("Rendering "+ fullpath)
    body = untangle(c,p)
    jinja_render(body, fullpath, c.vs)
