    from leo.core import leoGlobals as g

    ok = g.app.gui.guiName() == "qt"
    if not ok:
        return False

    from leo.plugins import qtGui    
    setattr(qtGui.leoQtFrame, 'setTopGeometry', setTopGeometry_mod_framesize)
    g.plugin_signon(__name__)
    return True
